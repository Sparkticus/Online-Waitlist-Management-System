The Online Waitlist Management System (Part II)

Proposed by: Joanna Bi & Lindsey Tang

Design

Student Operations:

  1. Create accounts (info required: name, banner id, major/minor, class year, and email address)
  2. Search for specific waitlists
  3. Submit forms to be on a waitlist (w/ short explanation of about 50 words)
  4. View personal rank on waitlists
  5. Remove themselves from a specific waitlist
  6. When requested from prof, check in online to secure continued placement on waitlist (implement should time permit)

Professor Operations:

  1. Create accounts (info required: name, banner id, department and email address)
  2. Create a waitlist (info required: lab or class, crn, course number, department, enrollment limit)
  3. View their created waitlists and relevant info about students on those waitlists
  4. Adjust rankings of students on selected waitlists
  5. Select which waitlisted students to include in the lottery, then press a button to start the lottery (output a new list of selected students in random order, ask to confirm new rankings before storing lottery results to database)
  6. Collectively view email addresses of all students on a selected waitlist
  7. Start the online check-ins for students on waitlist (implement should time permit)

SQL Statements

drop table if exists Professor;

drop table if exists Students;

drop table if exists Created_Waitlists;

drop table if exists On_Waitlists;

drop table if exists Person;

drop table if exists Lab;

drop table if exists Course;

drop table if exists Class;

drop table if exists Waitlists;

drop table if exists Corresponding_Labs;

-- define the user table

create table Person(

        bid char(9) not null primary key,

        name varchar(30) not null,

        email varchar(30) not null,

        username varchar(20) not null,

        unique (username),

        pass varchar(20) not null

        )

        ENGINE = InnoDB;

--define the student table

create table Student(

        student_id char(9) primary key,

        foreign key (student_id) references Person (bid) on delete cascade on update cascade,

        class_year int(4) unsigned not null,

        major_minor varchar(30) not null

        )

        ENGINE = InnoDB;

-- define the professor table

create table Professor(

        bid char(9) primary key,

        foreign key (bid)

                references Person (bid)

                on delete cascade on update cascade,

        department varchar(10) not null

        )

        ENGINE = InnoDB;

-- define the course table

create table Course(

        crn int(5) not null primary key,

        course_num  varchar(10) not null,

        department varchar(20) not null,

        course_limit int unsigned not null

        )

        ENGINE = InnoDB;

--define the class table

create table Class(

        crn int(5) primary key,

        foreign key (crn)

                references Course (crn)

                on delete cascade on update cascade

        )

        ENGINE = InnoDB;

-- define the lab table

create table Lab(

        crn int(5) primary key,

        foreign key (crn)

                references Course (crn)

                on delete cascade on update cascade

        )

        ENGINE = InnoDB;

-- define the course-to-lab relationship (one to many)          

create table Corresponding_Labs(

        class_crn int not null,

        lab_crn int not null,

        primary key (class_crn, lab_crn),

        foreign key(class_crn) references class (crn) ,

        foreign key(lab_crn) references lab (crn)

        )

        ENGINE = InnoDB;

-- define the waitlist table

create table Waitlist(

        waitlist_id int unsigned not null primary key,

        student_bid char(9) not null,

        foreign key (student_bid)

                references Person (bid)

                on delete cascade on update cascade,

        student_name varchar(20) not null,

        foreign key (student_name )

                references Person (name)

                on delete cascade on update cascade,

        major_minor varchar(20) not null,

        foreign key (major_minor)

                references Student (major_minor)

                on delete cascade on update cascade,

        student_class int(4) not null,

        foreign key (student_class)

                references Student (class_year)

                on delete cascade on update cascade,

        submitted_on timestamp default current_timestamp not null,

        rank int unsigned not null,

        explanation varchar(250) default 'na'

        )

        ENGINE = InnoDB;

-- define the “student has waitlists” table

create table On_Waitlists(

        bid char(9) not null,

        waitlist_id int,

        primary key (bid, waitlist_id),

        foreign key (bid)

                references Person (bid)

                on delete cascade on update cascade,

        foreign key (waitlist_id)

                references Waitlist (waitlist_id)

                on delete cascade on update cascade

        )

        ENGINE = InnoDB;

-- define the “professor manages waitlists” table

create table Created_Waitlists(

        bid char(9),

        crn int(5),

        primary key (bid, crn),

        foreign key (bid)

                references Person (bid)

                on delete cascade on update cascade,

        foreign key (crn)

                references Course (crn)

                on delete cascade on update cascade

        )

        ENGINE = InnoDB;
